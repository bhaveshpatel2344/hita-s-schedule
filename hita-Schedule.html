<script>

let state = JSON.parse(localStorage.getItem('hita_shifts')) || {};

let activeKey = null; // "YYYY-MM-DD-job1"

const DOW = ['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];


function renderSchedule() {

    const startInput = document.getElementById('startDate').value;

    if(!startInput) return;

    

    const start = new Date(startInput + 'T00:00:00');

    const container = document.getElementById('scheduleGrid');

    container.innerHTML = '';

    

    const j1n = document.getElementById('j1Name').value;

    const j2n = document.getElementById('j2Name').value;

    

    let shiftCount = 0;


    for(let i=0; i<31; i++) {

        const d = new Date(start);

        d.setDate(d.getDate() + i);

        const key = d.toISOString().split('T')[0];

        

        const row = document.createElement('div');

        row.className = 'day-card';

        

        const s1 = state[key + '-j1'];

        const s2 = state[key + '-j2'];

        if(s1) shiftCount++;

        if(s2) shiftCount++;


        row.innerHTML = `

            <div class="day-info">

                <div class="dow">${DOW[d.getDay()]}</div>

                <div class="dom">${d.getDate()}</div>

            </div>

            <div class="shift-slot ${s1 ? 'filled j1-bg' : 'empty'}" onclick="openModal('${key}-j1')">

                ${s1 ? `<div class="s-name">${s1.label}</div><div class="s-time">${s1.start}-${s1.end}</div>` : `+ ${j1n}`}

            </div>

            <div class="shift-slot ${s2 ? 'filled j2-bg' : 'empty'}" onclick="openModal('${key}-j2')">

                ${s2 ? `<div class="s-name">${s2.label}</div><div class="s-time">${s2.start}-${s2.end}</div>` : `+ ${j2n}`}

            </div>

        `;

        container.appendChild(row);

    }

    document.getElementById('statCounter').innerText = `${shiftCount} Shifts This Period`;

    localStorage.setItem('hita_shifts', JSON.stringify(state));

}


function openModal(key) {

    activeKey = key;

    const shift = state[key];

    const isJ1 = key.includes('j1');

    const jobName = isJ1 ? document.getElementById('j1Name').value : document.getElementById('j2Name').value;

    

    document.getElementById('modalTitle').innerText = jobName;

    document.getElementById('customName').value = shift ? shift.label : '';

    document.getElementById('customStart').value = shift ? shift.start : '09:00';

    document.getElementById('customEnd').value = shift ? shift.end : '17:00';

    

    document.getElementById('modal').classList.add('active');

}


function applyPreset(label, s, e) {

    document.getElementById('customName').value = label;

    document.getElementById('customStart').value = s;

    document.getElementById('customEnd').value = e;

}


function cloneYesterday() {

    const parts = activeKey.split('-'); // [YYYY, MM, DD, job]

    const currentDay = new Date(parts[0], parts[1]-1, parts[2]);

    currentDay.setDate(currentDay.getDate() - 1);

    const yesterdayKey = currentDay.toISOString().split('T')[0] + '-' + parts[3];

    

    if(state[yesterdayKey]) {

        const prev = state[yesterdayKey];

        applyPreset(prev.label, prev.start, prev.end);

    } else {

        alert("No shift found for yesterday to copy!");

    }

}


function saveShift() {

    state[activeKey] = {

        label: document.getElementById('customName').value || 'Shift',

        start: document.getElementById('customStart').value,

        end: document.getElementById('customEnd').value

    };

    closeModal();

    renderSchedule();

}


function deleteShift() {

    delete state[activeKey];

    closeModal();

    renderSchedule();

}


function closeModal() {

    document.getElementById('modal').classList.remove('active');

}


function exportICS() {

    let ics = ['BEGIN:VCALENDAR', 'VERSION:2.0', 'PRODID:-//Hita//Schedule//EN'];

    for (let key in state) {

        const dateStr = key.split('-').slice(0,3).join('');

        const shift = state[key];

        const sTime = shift.start.replace(':', '') + '00';

        const eTime = shift.end.replace(':', '') + '00';

        

        ics.push('BEGIN:VEVENT');

        ics.push(`DTSTART:${dateStr}T${sTime}`);

        ics.push(`DTEND:${dateStr}T${eTime}`);

        ics.push(`SUMMARY:${shift.label}`);

        ics.push('END:VEVENT');

    }

    ics.push('END:VCALENDAR');

    

    const blob = new Blob([ics.join('\r\n')], {type: 'text/calendar'});

    const url = URL.createObjectURL(blob);

    const a = document.createElement('a');

    a.href = url;

    a.download = 'schedule.ics';

    a.click();

}


// Init

const now = new Date();

document.getElementById('startDate').value = now.toISOString().split('T')[0];

renderSchedule();
