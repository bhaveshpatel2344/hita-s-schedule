<script>

const STORAGE_KEY = "hita_schedule_v1";


let state = {};


const shifts = ["","Day","Evening","Night"];


function pad(n){ return String(n).padStart(2,"0"); }


function dateKey(d){

  return d.getFullYear()+"-"+pad(d.getMonth()+1)+"-"+pad(d.getDate());

}


function getDays(){

  const val = document.getElementById("monthPicker").value;

  if(!val) return [];


  const [y,m] = val.split("-").map(Number);

  const days = [];

  const count = new Date(y, m, 0).getDate();


  for(let i=1;i<=count;i++){

    days.push(new Date(y, m-1, i));

  }


  return days;

}


function build(){

  const grid = document.getElementById("grid");

  grid.innerHTML = "";


  const days = getDays();


  days.forEach(d=>{

    const key = dateKey(d);


    if(!state[key]) state[key] = 0;


    const div = document.createElement("div");

    div.className = "day";


    div.innerHTML = `

      <div>${d.getDate()}</div>

      <div class="shift">${shifts[state[key]]}</div>

    `;


    div.onclick = ()=>{

      state[key] = (state[key]+1) % shifts.length;

      save();

      build();

    };


    grid.appendChild(div);

  });

}


function save(){

  localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

}


function load(){

  const data = localStorage.getItem(STORAGE_KEY);

  if(data) state = JSON.parse(data);

}


function shareSchedule(){

  const encoded = btoa(JSON.stringify(state));

  const url = location.origin + location.pathname + "?data="+encoded;


  navigator.clipboard.writeText(url);

  alert("Link copied!");

}


function loadFromURL(){

  const params = new URLSearchParams(location.search);

  const data = params.get("data");

  if(data){

    state = JSON.parse(atob(data));

  }

}


(function init(){

  const d = new Date();

  document.getElementById("monthPicker").value =

    d.getFullYear()+"-"+pad(d.getMonth()+1);


  loadFromURL();

  load();

  build();

})();

</script>
